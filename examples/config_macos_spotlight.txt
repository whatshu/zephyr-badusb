REM ============================================
REM BadUSB Script: macOS Spotlight & Terminal
REM Opens Spotlight and runs Terminal commands
REM ============================================

DEFAULT_DELAY 100
DELAY 500

REM Open Spotlight (Command + Space)
GUI SPACE
DELAY 500

REM Type Terminal
STRING Terminal
DELAY 500
ENTER
DELAY 1000

REM Run commands
STRING echo "Hello from BadUSB on macOS!"
ENTER
DELAY 200

STRING sw_vers
ENTER
DELAY 200

STRING whoami
ENTER
